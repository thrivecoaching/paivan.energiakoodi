<script>
  const TOTAL = 10;
  let current = 1;
  const chips = {};

  function toggleChip(el, field) {
    el.classList.toggle('active');
    if (!chips[field]) chips[field] = [];
    const v = el.textContent.trim();
    chips[field] = chips[field].includes(v)
      ? chips[field].filter(x => x !== v)
      : [...chips[field], v];
  }

  function getChips(f) { return (chips[f] || []).join(', ') || 'ei valittu'; }

  function updateProgress(step) {
    document.getElementById('step-label').textContent = `Kysymys ${step}`;
    document.getElementById('step-fraction').textContent = `${step} / ${TOTAL}`;
    document.getElementById('progress-fill').style.width = `${(step / TOTAL) * 100}%`;
  }

  function goNext(from) {
    if (from >= TOTAL) return;
    const curr = document.getElementById(`q${from}`);
    const next = document.getElementById(`q${from + 1}`);
    curr.classList.remove('active');
    curr.classList.add('exit');
    setTimeout(() => { curr.classList.remove('exit'); }, 320);
    next.classList.add('active');
    current = from + 1;
    updateProgress(current);
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  function goBack(from) {
    if (from <= 1) return;
    const curr = document.getElementById(`q${from}`);
    const prev = document.getElementById(`q${from - 1}`);
    curr.classList.remove('active');
    prev.classList.add('active');
    current = from - 1;
    updateProgress(current);
    window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  async function analyze() {
    const btn = document.getElementById('analyzeBtn');
    const spinner = document.getElementById('btn-spinner');
    const btnText = document.getElementById('btn-text');

    btn.classList.add('loading');
    spinner.style.display = 'inline-block';
    btnText.textContent = 'Analysoidaan...';

    const fyysinen = getChips('fyysinen');
    const uni = getChips('uni');
    const tunnetila = getChips('tunnetila');
    const mentaali = document.getElementById('mentaali').value;
    const keho = document.getElementById('keho').value.trim() || 'ei mainittu';
    const paiva = document.getElementById('paiva').value.trim() || 'ei mainittu';
    const muu = document.getElementById('muu').value.trim() || 'ei lisättävää';
    const vuorovaikutus = getChips('vuorovaikutus');
    const tila = getChips('tila');
    const tarve = getChips('tarve');

    const prompt = `Olet asiantuntija hermoston säätelyssä, joogassa, somaattisessa tietoisuudessa ja arjen käyttäytymismalleissa.

Tehtäväsi on analysoida käyttäjän TÄMÄNPÄIVÄINEN TILA — ei persoonallisuus, ei pitkäaikaiset kaavat — VAIN TÄMÄ PÄIVÄ.

Käyttäjän vastaukset:
- Fyysinen tila: ${fyysinen}
- Unen laatu: ${uni}
- Tunnetila: ${tunnetila}
- Mentaalinen terävyys (1–10): ${mentaali}/10
- Kehon jännitys/raskaus: ${keho}
- Päivän sisältö: ${paiva}
- Muuta mielessä: ${muu}
- Vuorovaikutus tänään: ${vuorovaikutus}
- Vapaa tila päivässä: ${tila}
- Mitä tarvitsee eniten: ${tarve}

Anna analyysi TÄSMÄLLEEN tässä rakenteessa ja VAIN suomeksi. Käytä selkeää, suoraa ja rauhallista kieltä. Ei geneeristä neuvontaa. Kaikki täytyy tuntua henkilökohtaiselta ja spesifiltä juuri tälle päivälle.

Vastaa VAIN seuraavassa JSON-muodossa ilman mitään muuta tekstiä:
{
  "energiatila": "Kuvaa selkeästi miten hermosto ja energia toimivat tänään. Konkreettinen, ei abstrakti. (2-4 lausetta)",
  "riski": "Mikä on todennäköisin kaava joka heittää tämän päivän raiteiltaan? Spesifi tälle päivälle. (1-3 lausetta)",
  "prioriteetti": "YKSI selkeä suunta koko päivälle. Ei tehtävälista — ohjaava periaate. (1-2 lausetta, vahva)",
  "toimenpiteet": ["Toimenpide 1 — konkreettinen ja helppo toteuttaa tässä päivässä", "Toimenpide 2 — konkreettinen ja helppo toteuttaa tässä päivässä", "Toimenpide 3 — konkreettinen ja helppo toteuttaa tässä päivässä"],
  "tietoisuus": "Yksi lyhyt, terävä havainto siitä mitä heidän täytyy tänään huomata itsessään. (1 lause)"
}`;

    try {
      const response = await fetch("/api/analyze", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          messages: [{ role: "user", content: prompt }]
        })
      });

      const data = await response.json();
      const text = data.content.map(i => i.text || '').join('');
      const clean = text.replace(/```json|```/g, '').trim();
      const parsed = JSON.parse(clean);
      renderResult(parsed);
    } catch (err) {
      btn.classList.remove('loading');
      spinner.style.display = 'none';
      btnText.textContent = 'Analysoi päivä →';
      alert('Virhe analyysissä. Yritä uudelleen.');
      console.error(err);
    }
  }

  function renderResult(data) {
    document.getElementById('quiz-section').style.display = 'none';
    document.getElementById('main-header').style.display = 'none';

    const result = document.getElementById('result');
    result.style.display = 'block';

    const now = new Date();
    document.getElementById('result-date').textContent =
      now.toLocaleDateString('fi-FI', { weekday: 'long', day: 'numeric', month: 'long' });

    document.getElementById('result-cards').innerHTML = `
      <div class="result-card">
        <div class="result-label">🔍 Tänään energiatila</div>
        <div class="result-body">${data.energiatila}</div>
      </div>
      <div class="result-card">
        <div class="result-label">⚠️ Tänään riski</div>
        <div class="result-body">${data.riski}</div>
      </div>
      <div class="result-card priority">
        <div class="result-label">🎯 Tänään prioriteetti</div>
        <div class="result-body">${data.prioriteetti}</div>
      </div>
      <div class="result-card">
        <div class="result-label">🧘 3 toimenpidettä tänään</div>
        <div class="actions-list">
          ${data.toimenpiteet.map((a, i) => `
            <div class="action-item">
              <span class="action-num">${i + 1}</span>
              <span>${a}</span>
            </div>`).join('')}
        </div>
      </div>
      <div class="result-card awareness">
        <div class="result-label">💡 Tietoisuus</div>
        <div class="result-body">${data.tietoisuus}</div>
      </div>`;

    window.scrollTo({ top: 0, behavior: 'smooth' });
  }

  function resetAll() {
    // Reset chips
    Object.keys(chips).forEach(k => chips[k] = []);
    document.querySelectorAll('.chip.active').forEach(c => c.classList.remove('active'));

    // Reset inputs
    document.getElementById('mentaali').value = 5;
    document.getElementById('mentaali-val').textContent = '5';
    document.getElementById('keho').value = '';
    document.getElementById('paiva').value = '';
    document.getElementById('muu').value = '';

    // Reset slides
    document.querySelectorAll('.q-slide').forEach(s => s.classList.remove('active', 'exit'));
    document.getElementById('q1').classList.add('active');
    current = 1;
    updateProgress(1);

    // Reset btn
    const btn = document.getElementById('analyzeBtn');
    btn.classList.remove('loading');
    document.getElementById('btn-spinner').style.display = 'none';
    document.getElementById('btn-text').textContent = 'Analysoi päivä →';

    // Show form, hide result
    document.getElementById('main-header').style.display = 'block';
    document.getElementById('quiz-section').style.display = 'block';
    document.getElementById('result').style.display = 'none';

    window.scrollTo({ top: 0, behavior: 'smooth' });
  }
</script>
